#make a function that converts lists of state abbreviations to lists of state names
to_statename <- function(list){
  new_list <- c()
  for(i in 1:length(list)){
    name <- state.name[grep(list[i], state.abb)] 
    new_list <- append(new_list, name)
  }
  new_list
}
